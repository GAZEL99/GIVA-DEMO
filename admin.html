Setup Database — SQL Editor

Cara pakai: Buka Supabase Dashboard → pilih project → klik SQL Editor → paste SQL di bawah → klik Run. Jalankan masing-masing blok secara terpisah.
Tabel: promo_codes 
CREATE TABLE IF NOT EXISTS public.promo_codes (
  id                  uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
  code                text        NOT NULL UNIQUE,
  discount_type       text        NOT NULL DEFAULT 'percentage'
                                  CHECK (discount_type IN ('percentage','fixed')),
  discount_value      numeric(10,2) NOT NULL DEFAULT 0,
  max_discount_amount numeric(15,0),
  min_order_amount    numeric(15,0) NOT NULL DEFAULT 0,
  max_usage           integer,
  used_count          integer     NOT NULL DEFAULT 0,
  valid_from          timestamptz,
  valid_until         timestamptz,
  description         text,
  is_active           boolean     NOT NULL DEFAULT true,
  created_at          timestamptz NOT NULL DEFAULT now()
);
ALTER TABLE public.promo_codes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "admin_promo_all"   ON public.promo_codes FOR ALL
  USING (auth.role() = 'authenticated') WITH CHECK (auth.role() = 'authenticated');
CREATE POLICY "anon_promo_read"   ON public.promo_codes FOR SELECT
  USING (is_active = true);
Tabel: site_popups 
CREATE TABLE IF NOT EXISTS public.site_popups (
  id                   uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
  title                text        NOT NULL,
  message              text        NOT NULL,
  image_url            text,
  button_text          text,
  button_url           text,
  popup_type           text        NOT NULL DEFAULT 'announcement',
  position             text        NOT NULL DEFAULT 'center',
  show_on_page         text        NOT NULL DEFAULT 'all',
  delay_seconds        integer     NOT NULL DEFAULT 3,
  bg_color             text        NOT NULL DEFAULT '#1E3A8A',
  is_active            boolean     NOT NULL DEFAULT true,
  is_closeable         boolean     NOT NULL DEFAULT true,
  show_once_per_session boolean    NOT NULL DEFAULT false,
  created_at           timestamptz NOT NULL DEFAULT now()
);
ALTER TABLE public.site_popups ENABLE ROW LEVEL SECURITY;
CREATE POLICY "admin_popup_all"   ON public.site_popups FOR ALL
  USING (auth.role() = 'authenticated') WITH CHECK (auth.role() = 'authenticated');
CREATE POLICY "anon_popup_read"   ON public.site_popups FOR SELECT
  USING (is_active = true);
